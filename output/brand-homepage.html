代码示例

通过一个简单的 AgentDSL 脚本了解其强大功能。


// 定义一个简单的问候代理
agent GreeterAgent:
    state:
        name: "World"

    on start:
        log("GreeterAgent 已启动！")

    on message msg:
        if msg.type == "greet":
            targetName = msg.payload.name or state.name
            send(msg.sender, "Hello, " + targetName + "!")
        else:
            log("收到未知消息: " + msg.type)

// 实例化并运行代理
run GreeterAgent as "myGreeter"
send("myGreeter", { "type": "greet", "payload": { "name": "AgentDSL User" } })
